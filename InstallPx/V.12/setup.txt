1. Copy .env.example  content into .env
2. add database name
3. create database
4. add composer.local.json file in base direcotry
{
    "require-dev": {
        "vpx/pxcommands": "1.0"
    },
    "repositories": [
        {
            "type": "path",
            "url": "./../vpx/pxcommands",
            "options": {
                "symlink": true
            }
        }
    ]
}
5. composer install
6. php artisan px:deploy --for=local --baseUrl=http://IP/revital-church/public/ --name=revital_church --user=root --pass= --smtp=default
7. php artisan migrate:fresh --seed
